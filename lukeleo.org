** 配置loadpath
#+BEGIN_SRC emacs-lisp
(setq default-directory "~/")
(add-to-list 'load-path "~/.emacs.d/luke/")

(defun open-init-file()
  (interactive)
  (find-file "~/.emacs.d/init.el"))
#+END_SRC
